create table FREE_BOARD
(
	FBOARD_NUM NUMBER(20) not null

	FTITLE VARCHAR2(100) not null,
	FCONTENT VARCHAR2(255),
	FDATE DATE default SYSDATE,
	FHIT NUMBER(10) default 0,
	FGROUP NUMBER(4),
	FSTEP NUMBER(4),
	FINDENT NUMBER(4),
	LIKE_CNT NUMBER default 0
)
